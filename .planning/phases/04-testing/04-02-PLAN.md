```python
class OrderType(Enum):
    MARKET_BUY = "market_buy"

    BUY_STOP = "buy_stop"

class OrderResult:
    success: bool; ticket: int = 0; price: float = 0.0; volume: float = 0.0; error: str = ""

class Position:
    ticket: int; symbol: str; direction: str; volume: float; open_price: float
    sl: float; tp: float; profit: float; comment: str = ""

class AccountInfo:
    balance: float; equity: float; margin: float; free_margin: float; currency: str = "USD"

class DryRunConnector(MT5Connector):
    _ticket_counter: int = 100000  # CLASS-level shared counter
    _fake_positions: dict[int, Position]  # Instance-level

    async def ping(self) -> bool  # Returns self._connected
    async def connect(self, password=None) -> bool  # Sets _connected=True, clears password
    async def disconnect(self) -> None  # Sets _connected=False
    async def get_price(self, symbol) -> None  # Always returns None!
    async def get_account_info(self) -> AccountInfo  # Returns $10k balance
    async def get_positions(self, symbol=None) -> list[Position]
    async def open_order(self, symbol, order_type, volume, price=0, sl=0, tp=0, comment="") -> OrderResult
    async def modify_position(self, ticket, sl=None, tp=None) -> OrderResult
    async def close_position(self, ticket, volume=None) -> OrderResult
    async def cancel_pending(self, ticket) -> OrderResult
    async def get_pending_orders(self, symbol=None) -> list[dict]  # Always returns []

def create_connector(backend, account_name, server, login, password, **kwargs) -> MT5Connector
```

```python
class FailingConnector(DryRunConnector):
    def __init__(self, *args, fail_on: set[str] | None = None, **kwargs):
        super().__init__(*args, **kwargs)
        self._fail_on = fail_on or set()

    async def ping(self) -> bool:
        if "ping" in self._fail_on:
            self._connected = False
            return False
        return await super().ping()

    async def get_price(self, symbol: str):
        if "get_price" in self._fail_on:
            return None
        return (4980.0, 4981.0)  # Override None default for tests needing prices

    async def open_order(self, *args, **kwargs):
        if "open_order" in self._fail_on:
            return OrderResult(success=False, error="Simulated failure")
        return await super().open_order(*args, **kwargs)
```
