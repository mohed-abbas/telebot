Ship SET-03 — the per-account settings form with a two-step dangerous-change modal and an audit-timeline revert button — on the dashboard.

Purpose: operators need a safe, auditable way to adjust per-account trading parameters at runtime without editing JSON files or restarting the bot.     Step 1 — Hard-cap validator module.
    Add a module-level helper to dashboard.py (or a new `settings_validators.py`):
    ```python
    from dataclasses import dataclass

    @dataclass
    class ValidationError:
        field: str
        message: str

    _HARD_CAPS = {
        "max_stages": (1, 10),
        "default_sl_pips": (1, 500),
        "max_daily_trades": (1, 100),
    }

    def validate_settings_form(
        form: dict,
        max_lot_size: float,  # per-account from accounts.json — for fixed_lot upper bound
    ) -> tuple[dict, list[ValidationError]]:
        """Returns (parsed_values_dict, [errors]). parsed_values empty if any error.

        Applies D-29 hard caps server-side. Caller is authoritative; client-echo
        via HTML min/max is cosmetic (D-29 Claude's discretion).
        """
        errors: list[ValidationError] = []
        parsed: dict = {}

        # risk_mode
        risk_mode = form.get("risk_mode", "").strip()
        if risk_mode not in ("percent", "fixed_lot"):
            errors.append(ValidationError("risk_mode", 'Risk mode must be "percent" or "fixed_lot".'))
        else:
            parsed["risk_mode"] = risk_mode

        # risk_value — cap depends on risk_mode
        try:
            risk_value = float(form.get("risk_value", ""))
        except (ValueError, TypeError):
            errors.append(ValidationError("risk_value", "Risk value must be a number."))
            risk_value = None
        if risk_value is not None:
            if risk_value <= 0:
                errors.append(ValidationError("risk_value", "Risk value must be greater than 0."))
            elif risk_mode == "percent" and risk_value > 5.0:
                errors.append(ValidationError("risk_value", "risk_value must be between 0 and 5.0."))
            elif risk_mode == "fixed_lot" and risk_value > max_lot_size:
                errors.append(ValidationError("risk_value", "Risk value exceeds max_lot_size for this account."))
            else:
                parsed["risk_value"] = risk_value

        # integer fields with simple (min, max) caps
        for field, (min_v, max_v) in _HARD_CAPS.items():
            raw = form.get(field, "")
            try:
                v = int(raw)
            except (ValueError, TypeError):
                errors.append(ValidationError(field, f"{field} must be an integer."))
                continue
            if v < min_v or v > max_v:
                errors.append(ValidationError(field, f"{field} must be between {min_v} and {max_v}."))
            else:
                parsed[field] = v

        return (parsed if not errors else {}), errors
    ```

    ```python
    @app.post("/settings/{account_name}", response_class=HTMLResponse)

    ):
        """D-27: validate hard caps; on success render modal HTML; on failure 422 with field errors."""
        form = dict(await request.form())
        # Look up max_lot_size from accounts.json / accounts table for this account.
        account_row = await _get_account_row(account_name)
        if account_row is None:
            raise HTTPException(status_code=404, detail=f"Unknown account: {account_name}")
        max_lot_size = account_row.get("max_lot_size", 1.0)

        parsed, errors = validate_settings_form(form, max_lot_size=max_lot_size)
        if errors:
            # Re-render the form partial with red-400 inline errors.
            current = _settings_store.effective(account_name)
            return templates.TemplateResponse(
                "partials/account_settings_tab.html",
                {
                    "request": request, "a": account_row, "s": current,
                    "errors": {e.field: e.message for e in errors},
                    "audit": await db.get_settings_audit(account_name, limit=50),
                },
                status_code=422,
            )

        # Compute diff vs current effective settings
        current = _settings_store.effective(account_name)
        diff = []
        for field, new_val in parsed.items():
            old_val = getattr(current, field)
            if str(old_val) != str(new_val):
                diff.append({"field": field, "old": old_val, "new": new_val})

        if not diff:
            # Nothing changed; bounce with a light message instead of the modal.
            return HTMLResponse('<div class="text-xs text-slate-500">No changes to save.</div>')

        # Dry-run preview (simple string; planner chooses math)
        dry_run = _compute_dry_run(parsed, current)

        return templates.TemplateResponse(
            "partials/settings_confirm_modal.html",
            {
                "request": request, "account_name": account_name,
                "diff": diff, "dry_run": dry_run, "pending_values": parsed,
            },
        )
    ```

    ```python
    def _compute_dry_run(parsed: dict, current) -> str:
        risk_mode = parsed.get("risk_mode", current.risk_mode)
        risk_value = parsed.get("risk_value", current.risk_value)
        max_stages = parsed.get("max_stages", current.max_stages)
    ```

    Step 4 — POST /settings/{account_name}/confirm.
    ```python
    @app.post("/settings/{account_name}/confirm", response_class=HTMLResponse)
    async def settings_confirm(
        account_name: str, request: Request,
        user: str = Depends(_verify_auth), _csrf=Depends(_verify_csrf),
    ):
        """D-27: persist validated settings. Writes audit row via SettingsStore.update."""
        form = dict(await request.form())
        account_row = await _get_account_row(account_name)
        if account_row is None:
            raise HTTPException(status_code=404)
        max_lot_size = account_row.get("max_lot_size", 1.0)
        parsed, errors = validate_settings_form(form, max_lot_size=max_lot_size)
        if errors:
            raise HTTPException(status_code=422, detail="Re-validation failed")

        for field, new_val in parsed.items():
            current = _settings_store.effective(account_name)
            if str(getattr(current, field)) != str(new_val):
                await _settings_store.update(account_name, field, new_val, actor=user)

        # Re-render the full tab partial with fresh audit row on top
        fresh = _settings_store.effective(account_name)
        audit = await db.get_settings_audit(account_name, limit=50)
        return templates.TemplateResponse(
            "partials/account_settings_tab.html",
            {"request": request, "a": account_row, "s": fresh, "errors": {}, "audit": audit},
        )
    ```

    ```python
    @app.post("/settings/{account_name}/revert", response_class=HTMLResponse)
    async def settings_revert(
        account_name: str, audit_id: int, request: Request,
        user: str = Depends(_verify_auth), _csrf=Depends(_verify_csrf),
    ):
        """D-28: re-open the two-step modal with the inverted diff pre-populated."""
        audit_row = await db.get_settings_audit_row(audit_id)  # fetch one row by id
        if audit_row is None or audit_row["account_name"] != account_name:
            raise HTTPException(status_code=404)
        # Show a modal asking operator to confirm revert
        current = _settings_store.effective(account_name)
        diff = [{
            "field": audit_row["field"],
            "old": getattr(current, audit_row["field"]),
            "new": audit_row["old_value"],  # reverting: new = old_value from audit
        }]
        # Inverted pending values
        pending = {audit_row["field"]: audit_row["old_value"]}
        dry_run = f"Reverting {audit_row['field']} from {current.__dict__[audit_row['field']]} back to {audit_row['old_value']}."
        return templates.TemplateResponse(
            "partials/settings_confirm_modal.html",
            {
                "request": request, "account_name": account_name,
                "diff": diff, "dry_run": dry_run, "pending_values": pending,
                "is_revert": True,
            },
        )
    ```
