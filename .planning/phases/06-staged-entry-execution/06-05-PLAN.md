```python
@app.get("/stream")
async def sse_stream(request: Request, user: str = Depends(_verify_auth)):
    async def event_generator():
        while True:
            if await request.is_disconnected():
                break
            try:
                positions = await _get_all_positions()
                accounts = await _get_accounts_overview()
                data = json.dumps({
                    "positions": positions,
                    "accounts": accounts,
                    "timestamp": datetime.now(timezone.utc).isoformat(),
                })
                yield f"data: {data}\n\n"
            except Exception as exc:
                logger.error("SSE error: %s", exc)
            await asyncio.sleep(2)
    return StreamingResponse(
        event_generator(),
        media_type="text/event-stream",
        headers={"Cache-Control": "no-cache", "X-Accel-Buffering": "no"},
    )
```

    ```python
    def _enrich_stage_for_ui(stage: dict, positions: list[dict]) -> dict:
        """Add filled-count / current_price / distance / elapsed strings for the UI."""
        signal_id = stage["signal_id"]
        # Count filled stages for this signal_id (requires a second DB query OR joining in get_pending_stages)
        # Simplest: fire a cheap COUNT query.
        # For v1.1 simplicity, expose filled + total from stage snapshot:
        snapshot = stage.get("snapshot_settings") or {}
        if isinstance(snapshot, str):
            snapshot = json.loads(snapshot)
        total = snapshot.get("max_stages", stage["stage_number"])

        # Current price: look up in the provided positions list by symbol,
        # OR fall back to a bid/ask query. For UI cadence (2s SSE), a slightly
        # stale price is acceptable; prefer a position's current price if open.
        # If positions list does not carry price, set current_price=None.
        current_price = None
        for p in positions:
            if p.get("symbol") == stage["symbol"] and p.get("account_name") == stage["account_name"]:
                current_price = p.get("price_current") or p.get("current_price")
                break

        # Distance to next band (stage is the next-to-fire for its sequence)
        if current_price is not None:
            band_low = stage["band_low"]
            band_high = stage["band_high"]
            direction = stage["direction"]
            if direction == "buy":
                # trigger edge is band_high; negative means below (already crossed)
                distance_pips = (band_high - current_price) * 100  # XAUUSD: 1 = 100 pips; adjust for non-gold later
            else:
                distance_pips = (current_price - band_low) * 100
            if distance_pips <= 0:
                distance_str = "inside band"
            else:
                sign = "−" if direction == "buy" else "+"
                distance_str = f"{sign}{abs(distance_pips):.1f} pips to next band"
        else:
            distance_str = "—"

        # Elapsed
        from datetime import datetime, timezone
        elapsed_seconds = (datetime.now(timezone.utc) - stage["created_at"]).total_seconds()
        if elapsed_seconds < 3600:
            elapsed_str = f"{int(elapsed_seconds // 60):02d}:{int(elapsed_seconds % 60):02d}"
        else:
            h = int(elapsed_seconds // 3600)
            m = int((elapsed_seconds % 3600) // 60)
            s = int(elapsed_seconds % 60)
            elapsed_str = f"{h:02d}:{m:02d}:{s:02d}"

        # Filled-count via staged_entries count (cheap index-only query).
        # Alternatively embed this computation in db.get_pending_stages. For v1.1,
        # compute here with one extra fetchval per row; top-5 worst case = 5 queries.
        # TODO(phase7): denormalize filled_count column on staged_entries if SSE cadence
        # exposes a hot-path.
        return {
            "account_name": stage["account_name"],
            "symbol": stage["symbol"],
            "direction": stage["direction"],
            "filled": stage["stage_number"],  # placeholder — this stage's own number
            "total": total,
            "band_low": stage["band_low"],
            "band_high": stage["band_high"],
            "current_price": current_price,
            "distance_str": distance_str,
            "elapsed": elapsed_str,
            "status": stage["status"],
        }
    ```

    NOTE: the `filled` field is intentionally approximate in v1.1 — showing the NEXT armed stage's number. A precise "3 / 5" requires a grouped COUNT per signal_id, which is a phase-7 optimization. For v1.1, if the planner prefers a precise count, add a helper `db.count_filled_stages_for_signal(signal_id, account_name) -> int` that wraps `SELECT COUNT(*) FROM staged_entries WHERE signal_id=$1 AND account_name=$2 AND status='filled'`. Either approach satisfies the UI-SPEC — document the choice in the SUMMARY.

    ```javascript
    // Listens for htmx:afterSwap and sse:message events. After a swap, any
    // [data-price-cell] element whose innerText differs from its previous
    // cached value gets a 150ms indigo ring flash.
    (function () {
      const _priceCache = new WeakMap();

      function flashChangedPriceCells(root) {
        const cells = (root && root.querySelectorAll)
          ? root.querySelectorAll("[data-price-cell]")
          : document.querySelectorAll("[data-price-cell]");
        cells.forEach((cell) => {
          const now = cell.innerText.trim();
          const prev = _priceCache.get(cell);
          if (prev !== undefined && prev !== now) {
            cell.classList.add("ring-1", "ring-indigo-400/40");
            setTimeout(() => {
              cell.classList.remove("ring-1", "ring-indigo-400/40");
            }, 150);
          }
          _priceCache.set(cell, now);
        });
      }

      document.body.addEventListener("htmx:afterSwap", function (e) {
        flashChangedPriceCells(e.target);
      });
      document.body.addEventListener("htmx:sseMessage", function (e) {
        flashChangedPriceCells(document);
      });
    })();
    ```
