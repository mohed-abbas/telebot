Land the Executor-side safety wiring: `_zone_watch_loop` (D-11/D-14) that polls MT5 price and fires armed stages when a price band is entered, plus `_trading_paused` mid-tick re-check; `emergency_close` extended to drain `staged_entries` BEFORE the position-close loop (D-21); `_sync_positions` extended to reconcile staged entries on reconnect by MT5 `comment` prefix (D-24/D-25) with an idempotency probe that prevents double-submit; plus the **D-16 sequence-lifetime-equals-stage-1** invariant — before firing any stage 2..N, verify stage 1's MT5 position is still open; if not, cancel all remaining unfilled stages for that signal_id.

Purpose: every failure mode that operators trust v1.0 to handle (kill switch instantly stops new fills, reconnect reconciles state without duplicating positions, stage 1 exit terminates the scaling-in plan) must also hold for staged entries. This plan is the most-safety-critical code in Phase 6 — it reuses trade_manager.py's stage-aware `_execute_open_on_account` (from Plan 02) as the single fill path so the invariants already tested there carry through.

    Step 2 — Add `_zone_watch_loop` method. Structure mirrors `_heartbeat_loop` exactly. The D-16 pre-fire cascade sits between the in-zone check and the idempotency probe:
    ```python
    async def _zone_watch_loop(self) -> None:
        """D-11/D-14 — poll MT5 price for every awaiting_zone stage; fire when band entered.

        Cadence: 10 seconds uniform. Cross-cuts:
          - D-21 kill-switch: loop-entry guard + per-stage tick re-check between pre-flight and submit
          - D-24/D-25 idempotency: probe MT5 by target comment before any submit
          - D-14 pre-flight: re-fetch bid/ask; require within band ± 0.5*band_width
          - D-16 cascade: if stage 1 of this signal_id is no longer open on MT5, cancel remaining stages
        """
        import json
        from models import SignalAction, SignalType, Direction, AccountSettings
        from trade_manager import compute_bands, stage_is_in_zone_at_arrival, Band

        # Per-loop-tick cache of (acct_name, signal_id) -> bool (True if stage 1 still open)
        # Rebuilt each tick; avoids redundant MT5 scans when multiple stages share the same parent.

        while True:
            try:
                await asyncio.sleep(10)  # D-14 cadence
                if self._trading_paused:
                    continue

                rows = await db.get_active_stages()
                if not rows:
                    continue

                by_pair: dict[tuple[str, str], list[dict]] = {}
                for r in rows:
                    by_pair.setdefault((r["account_name"], r["symbol"]), []).append(r)

                # D-16 cache — per-tick memoization of stage-1 liveness per (account, signal_id)
                stage1_live_cache: dict[tuple[str, int], bool] = {}

                for (acct_name, symbol), stages in by_pair.items():
                    if acct_name in self._reconnecting:
                        continue
                    connector = self.tm.connectors.get(acct_name)
                    if connector is None:
                        continue
                    try:
                        bid, ask = await connector.get_price(symbol)
                    except Exception as exc:
                        logger.warning("%s: zone-watch get_price failed: %s", acct_name, exc)
                        continue

                    # Pre-fetch positions once per (account, symbol) — reused for idempotency probe + D-16 check
                    try:
                        positions = await connector.get_positions()
                    except Exception as exc:
                        logger.warning("%s: get_positions failed in zone-watch: %s", acct_name, exc)
                        continue

                    # Build comment index (used by BOTH D-16 stage-1 check and D-25 idempotency probe)
                    positions_by_comment = {
                        getattr(p, "comment", ""): p for p in positions if getattr(p, "comment", "")
                    }

                    for stage in stages:
                        # D-21: re-check INSIDE per-stage tick
                        if self._trading_paused:
                            break

                        signal_id = stage["signal_id"]
                        band_low = stage["band_low"]
                        band_high = stage["band_high"]
                        band_width = band_high - band_low
                        tolerance = 0.5 * band_width
                        direction = stage["direction"]

                        # Has price entered the band?
                        band = Band(stage_number=stage["stage_number"], low=band_low, high=band_high)
                        if not stage_is_in_zone_at_arrival(band, bid, ask, direction):
                            continue

                        # D-16 — stage-1-exit cascade. Verify stage 1 for this signal_id is still open on MT5.
                        # Cached per (acct_name, signal_id) to avoid N MT5 scans when multiple stages share a parent.
                        cache_key = (acct_name, signal_id)
                        if cache_key not in stage1_live_cache:
                            stage1_comment = f"telebot-{signal_id}-s1"
                            stage1_live = stage1_comment in positions_by_comment
                            stage1_live_cache[cache_key] = stage1_live
                            if not stage1_live:
                                # Verify stage 1 was ever filled (not just "never fired"). If the DB row
                                # for stage 1 is status='filled' but position absent, that means exit.
                                try:
                                    stage1_row = await db.get_stage_by_comment(stage1_comment)
                                except Exception:
                                    stage1_row = None
                                if stage1_row and stage1_row.get("status") == "filled":
                                    # D-16 cascade fires: cancel remaining unfilled stages for this signal_id.
                                    cancelled = await db.cancel_unfilled_stages_for_signal(
                                        signal_id, reason="stage1_closed",
                                    )
                                    logger.info(
                                        "%s: D-16 cascade — stage 1 of signal_id=%d no longer open on MT5; "
                                        "cancelled %d unfilled stage(s)",
                                        acct_name, signal_id, cancelled,
                                    )
                                    # Mark this signal as "do not fire" for the rest of this tick
                                    # (the DB update means subsequent get_active_stages() won't return them anyway,
                                    # but we're mid-tick so we still need the guard).
                                    continue
                                else:
                                    # Stage 1 not filled yet (still awaiting) — do NOT cascade; let things be.
                                    # This handles the edge case where the text-only signal is in flight
                                    # but stage 1 hasn't completed. Cache accordingly.
                                    stage1_live_cache[cache_key] = True

                        if not stage1_live_cache[cache_key]:
                            continue  # subsequent stages for this already-cascaded signal_id

                        # D-14 pre-flight re-check
                        try:
                            bid2, ask2 = await connector.get_price(symbol)
                        except Exception:
                            continue
                        price_center = (band_low + band_high) / 2
                        mid_price = (bid2 + ask2) / 2
                        if abs(mid_price - price_center) > band_width + tolerance:
                            logger.info(
                                "%s: zone-watch skip stage=%d signal_id=%d — price drifted outside band±tolerance",
                                acct_name, stage["stage_number"], signal_id,
                            )
                            continue

                        # D-21 re-check BEFORE submit
                        if self._trading_paused:
                            break

                        # D-25 idempotency probe: if MT5 already has this comment, mark filled.
                        target_comment = stage["mt5_comment"]
                        match = positions_by_comment.get(target_comment)
                        if match:
                            await db.update_stage_status(stage["id"], "filled", mt5_ticket=match.ticket)
                            logger.info(
                                "%s: zone-watch idempotency match — stage=%d already on MT5 ticket=%d",
                                acct_name, stage["stage_number"], match.ticket,
                            )
                            continue

                        # Build synthetic signal and call stage-aware _execute_open.
                        snapshot_dict = stage["snapshot_settings"]
                        if isinstance(snapshot_dict, str):
                            snapshot_dict = json.loads(snapshot_dict)
                        snapshot = AccountSettings(**snapshot_dict)

                        pip_value = 0.01 if symbol.upper().startswith("XAU") else 0.0001
                        entry_price = band_high if direction == "buy" else band_low
                        if direction == "buy":
                            sl_price = entry_price - snapshot.default_sl_pips * pip_value
                        else:
                            sl_price = entry_price + snapshot.default_sl_pips * pip_value

                        synthetic = SignalAction(
                            type=SignalType.OPEN,
                            symbol=symbol,
                            direction=Direction.BUY if direction == "buy" else Direction.SELL,
                            entry_zone=(band_low, band_high),
                            sl=sl_price,
                            tps=[],
                            raw_text=f"<zone-watch stage {stage['stage_number']} signal_id={signal_id}>",
                        )

                        try:
                            result = await self.tm._execute_open_on_account(
                                name=acct_name,
                                connector=connector,
                                signal=synthetic,
                                positions=positions,
                                signal_id=signal_id,
                                stage_number=stage["stage_number"],
                                stage_row_id=stage["id"],
                                snapshot=snapshot,
                            )
                        except Exception as exc:
                            logger.error(
                                "%s: zone-watch stage fill raised: %s (signal_id=%d stage=%d)",
                                acct_name, exc, signal_id, stage["stage_number"],
                            )
                            await db.update_stage_status(stage["id"], "failed", cancelled_reason=str(exc)[:200])
                            continue

                        if result.get("status") == "filled":
                            await db.update_stage_status(stage["id"], "filled", mt5_ticket=result.get("ticket"))
                        elif result.get("status") == "capped":
                            pass
                        elif result.get("status") in ("failed", "skipped"):
                            await db.update_stage_status(
                                stage["id"], "failed",
                                cancelled_reason=result.get("reason", "unknown")[:200],
                            )
            except asyncio.CancelledError:
                break
            except Exception as exc:
                logger.error("Zone watch loop error: %s", exc)
    ```

        """Full position sync from MT5 after reconnect (REL-02 + Phase 6 D-24/D-25)."""
        try:
            positions = await connector.get_positions()
            logger.info("%s: Position sync — %d open position(s)", acct_name, len(positions))
        except Exception as exc:
            logger.error("%s: Position sync failed: %s", acct_name, exc)
            return

        # D-24 — reconcile staged_entries against live MT5 positions by comment prefix
        try:
            pending = await db.get_pending_stages(account_name=acct_name)
        except Exception as exc:
            logger.error("%s: staged_entries fetch failed during sync: %s", acct_name, exc)
            return

        if not pending:
            return

        by_comment = {getattr(p, "comment", ""): p for p in positions if getattr(p, "comment", "")}

        reconciled = 0
        abandoned = 0
        for stage in pending:
            target_comment = stage["mt5_comment"]
            match = by_comment.get(target_comment)
            if match:
                await db.update_stage_status(stage["id"], "filled", mt5_ticket=match.ticket)
                reconciled += 1
                logger.info(
                    "%s: reconnect reconciled signal_id=%d stage=%d ticket=%d",
                    acct_name, stage["signal_id"], stage["stage_number"], match.ticket,
                )
                continue

            # No MT5 position. Is the stage old enough to give up on?
            max_age_minutes = await self._get_signal_max_age_minutes(stage["signal_id"])
            if max_age_minutes is None:
                max_age_minutes = getattr(self.tm.cfg, "max_age_minutes", 45)

            from datetime import datetime, timezone
            stage_age_minutes = (datetime.now(timezone.utc) - stage["created_at"]).total_seconds() / 60
            if stage_age_minutes > max_age_minutes:
                await db.update_stage_status(
                    stage["id"], "abandoned_reconnect",
                    cancelled_reason=f"no_mt5_match_after_{int(stage_age_minutes)}min",
                )
                abandoned += 1
                logger.warning(
                    "%s: reconnect abandoned signal_id=%d stage=%d — age=%dmin > %dmin and no MT5 position",
                    acct_name, stage["signal_id"], stage["stage_number"], int(stage_age_minutes), max_age_minutes,
                )

        if reconciled or abandoned:
            logger.info(
                "%s: staged_entries reconciliation — %d filled, %d abandoned",
                acct_name, reconciled, abandoned,
