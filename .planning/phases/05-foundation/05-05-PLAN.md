    Replace the current 27-line script with the following (exact content — portable POSIX sh constructs only; bash shebang retained for `set -euo pipefail`):

    ```bash
    #!/usr/bin/env bash
    # Tailwind build + Basecoat concat + content hash + manifest emit.
    # CWD must be repo root. Inputs: static/css/input.css + static/css/_compat.css + static/vendor/basecoat/basecoat.css + tailwind.config.js.
    # Outputs: static/css/app.<hash>.css + static/css/manifest.json.
    #
    # Why this is NOT a straight `tailwindcss -i input.css`:
    #   1. The Tailwind v3.4.19 standalone CLI does NOT resolve CSS @import directives,
    #      so `@import "./_compat.css"` and `@import "../vendor/basecoat/basecoat.css"`
    #      in input.css were silently dropped (UAT Gap #1).
    #   2. basecoat.css is authored for Tailwind v4 and contains `@custom-variant` / `@theme`
    #      at-rules that v3's CLI cannot parse — so it MUST be appended AFTER compilation,
    #      not piped through the CLI.
    #   3. _compat.css uses `@apply` and therefore MUST pass through Tailwind — we inline it
    #      into a temporary combined input so the Tailwind CLI sees its rules directly.
    set -euo pipefail

    INPUT="static/css/input.css"
    COMPAT="static/css/_compat.css"
    BASECOAT="static/vendor/basecoat/basecoat.css"
    COMBINED_IN="static/css/.build-input.css"      # tmp — inlined _compat, strip @import lines
    TAILWIND_OUT="static/css/.build-tailwind.css"  # tmp — Tailwind's output before basecoat append
    TMP_OUT="static/css/app.css"                   # tmp — final pre-hash

    # Clean any stale build artifacts so shell globs don't pick them up.
    rm -f static/css/app.*.css static/css/manifest.json "$COMBINED_IN" "$TAILWIND_OUT" "$TMP_OUT"

    # Build the combined Tailwind input: original input.css with @import lines stripped,
    # then the verbatim contents of _compat.css appended so @apply resolves.
    # The `@tailwind base/components/utilities` directives from input.css stay in place.
    grep -v '^@import' "$INPUT" > "$COMBINED_IN"
    cat "$COMPAT" >> "$COMBINED_IN"

    # Compile Tailwind (purges against tailwind.config.js content globs, resolves @apply in _compat).
    tailwindcss -i "$COMBINED_IN" -o "$TAILWIND_OUT" --minify

    # Append basecoat.css verbatim — its rules are already fully resolved CSS and must NOT
    # be fed through Tailwind v3 (see header comment, reason #2).
    cat "$TAILWIND_OUT" "$BASECOAT" > "$TMP_OUT"

    # Content-hash the final bundle.
    HASH=$(sha256sum "$TMP_OUT" | awk '{print substr($1,1,12)}')
    FINAL="static/css/app.${HASH}.css"
    mv "$TMP_OUT" "$FINAL"

    # Emit manifest pointing at the hashed filename.
    python3 - <<PY
    import json, pathlib
    pathlib.Path("static/css/manifest.json").write_text(json.dumps({
        "app.css": "app.${HASH}.css"
    }, indent=2))
    PY

    # Clean tmp intermediates (keep final hashed CSS + manifest).
    rm -f "$COMBINED_IN" "$TAILWIND_OUT"

    echo "Built: ${FINAL}"
    ```
