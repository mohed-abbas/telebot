```python
def _req(key: str, validator=None) -> str:
    val = environ.get(key)
    if not val:
        raise SystemExit(f"FATAL: Missing required env var: {key}")
    if validator and not validator(val):
        raise SystemExit(f"FATAL: Invalid format for {key}")

```
